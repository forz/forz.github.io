defer防止panic

利用 Commit() 之后 Rollback() 不会真正执行的特性,可以在创建tx时使用defer防止panic出现没有释放tx

package main

import (
    "database/sql"

    "x/log.v7"
)

func logRollback(tx *sql.Tx) {
    err := tx.Rollback()
    if err != tx.ErrTxDone && err != nil {
        log.Error()
    }
}


func main() {
    tx, err := sql.DB.Begin()
    if err != nil {
        log.Error()
        return
    }
    defer sql.Rollback(tx)
    //如果不添加defer，只在最后Commit后check错误err后再rollback，那么当doSomething发生异常的时候，函数就退出了，此时还没有执行到tx.Commit。这样就导致事务的连接没有关闭，事务也没有回滚。
    if result, err := tx.Exec(query1); err != nil {
        log.Error()
        return
    }
    doSomething()
    if row, err := tx.Query(query2); err != nil {
        //tx.Rollback() 没必要加,会自动进入defer
        log.Error()
        return
    }

    if err := tx.Commit(); err != nil {
        log.Error()
        return
    }
}

增加日志

官网文档用的时候就不判断错误，不判断错误如果rollback出错了，没日志，定位会很难，数据不一致问题也非常严重.但有一个弊端,就是即使commit成功后,依旧会进行rollback,并打印日志,此时应该加一个tag操作符,如果commit成功,tag置1,如果tag为1,不执行rollback

package main
import (
    "database/sql"

    "x/log.v7"
)

func logRollback(tx *sql.Tx) {
    err := tx.Rollback()
    if err != tx.ErrTxDone && err != nil {
        log.Error()
    }
}


func main() {
    tx, err := sql.DB.Begin()
    if err != nil {
        log.Error()
        return
    }
    defer logRollback(tx)
    if result, err := tx.Exec(query1); err != nil {
        log.Error()
        return
    }
    doSomething()
    if row, err := tx.Query(query2); err != nil {
        log.Error()
        return
    }

    if err := tx.Commit(); err != nil {
        log.Error()
        return
    }
}

我的写法

经过总结和实验，我采用了下面这种写法，defer tx.Rollback() 使得事务回滚始终得到执行。 当 tx.Commit() 执行后，tx.Rollback() 起到关闭事务的作用， 当程序因为某个错误中止，tx.Rollback() 起到回滚事务，同事关闭事务的作用。

      return
  }

  err = tx.Commit()
  return
}

LeetCode 118. Pascal’s Triangle#

Given numRows, generate the first numRows of Pascal’s triangle.

For example, given numRows = 5, Return

[
     [1],
    [1,1],
   [1,2,1],
  [1,3,3,1],
 [1,4,6,4,1]
]

解析：

杨辉三角形的构造方法：左右边界为1，中间num[i]=pre[i-1]+pre[i]

可以用递归来解

class Solution {
public:
    vector<vector<int> > generate(int numRows) {
        vector<int> vi;//每行
        vector<vector<int>> res;//整个杨辉三角形
        if (numRows == 0) {
            return res;
        }else if (numRows == 1) {//只有一行
            vi.push_back(1);
            res.push_back(vi);
        }else {
            res = generate(numRows - 1);//递归求得上一层的序列
            vector<int> tempv = res[numRows-2];//当前需构造行的前一行
            for (int i = 0; i < numRows; ++i) {//构造该行
                if (i == 0 || i == numRows-1) {//边界都是1
                    vi.push_back(1);
                } else {//内部
                    vi.push_back(tempv[i-1] + tempv[i]);
                }
            }
            res.push_back(vi);
        }
        return res;
    }
};

LeetCode 119. Pascal’s Triangle II#

Given an index k, return the kth row of the Pascal’s triangle.

For example, given k = 3,
Return [1,3,3,1].

Note: Could you optimize your algorithm to use only O(k) extra space?

解析：本题用递推方法，因为求解每一行只需要前一行的结果，所以可以用滚动数组

class Solution {  
public:  
vector<int> getRow(int rowIndex) {  // 只需要保存2行即可
    vector<int> result;  
    vector<int> tmp;  
    result.push_back(1);  //预处理第一行
    if (rowIndex <= 0)  
        return result;  
    for (int i=1; i<=rowIndex; i++)  //递推
    {  
        tmp=result; //保存前一行 
        result.clear();  
        result.push_back(1);  //头
        for (int i=0; i<tmp.size()-1; i++)  
        {  
            result.push_back(tmp[i] + tmp[i+1]);  
        }  
        result.push_back(1);  //尾

    }  
    return result;  
}  
};

LeetCode 70. Climbing Stairs

You are climbing a stair case. It takes n steps to reach to the top.

Each time you can either climb 1 or 2 steps. In how many distinct ways can you climb to the top?

Note: Given n will be a positive integer.

解析：标准Fibonacci，也可看作DP。f(n)=f(n-1)+f(n-2) 因为i只与i-1和i-2有关，可以滚动数组.

class Solution {
public:
int climbStairs(int n) {
    int prev = 0;//初始化边界值，因为1步=1，2步=2 只要接下来的递推有这两个初始值就可以
    int cur = 1;
    for(int i = 1; i <= n ; ++i){//从第一步开始枚举
        int tmp = cur;//局部变量
        cur += prev;//求和i
        prev = tmp;//维护i-2节点
    }
    return cur;
}
};

class Solution {
public:
int climbStairs(int n) {
    const double s = sqrt(5);
    return floor((pow((1+s)/2, n+1) + pow((1-s)/2, n+1))/s + 0.5);
}
};

#include <cstdio>
#include <iostream>

using namespace std;

const int MOD = 10000;

struct matrix
{
int m[2][2];
}ans, base;

matrix multi(matrix a, matrix b)
{
matrix tmp;
for(int i = 0; i < 2; ++i)
{
    for(int j = 0; j < 2; ++j)
    {
        tmp.m[i][j] = 0;
        for(int k = 0; k < 2; ++k)
            tmp.m[i][j] = (tmp.m[i][j] + a.m[i][k] * b.m[k][j]) % MOD;
    }
}
return tmp;
}
int fast_mod(int n)  // 求矩阵 base 的  n 次幂 
{
base.m[0][0] = base.m[0][1] = base.m[1][0] = 1;
base.m[1][1] = 0;
ans.m[0][0] = ans.m[1][1] = 1;  // ans 初始化为单位矩阵 
ans.m[0][1] = ans.m[1][0] = 0;
while(n)
{
    if(n & 1)  //实现 ans *= t; 其中要先把 ans赋值给 tmp，然后用 ans = tmp * t 
    {
        ans = multi(ans, base);
    }
    base = multi(base, base);
    n >>= 1;
}
return ans.m[0][1];
}

int main()
{
int n;
while(scanf("%d", &n) && n != -1)
{   
    printf("%d\n", fast_mod(n));
}
return 0;

LeetCode 91. Decode Ways

A message containing letters from A-Z is being encoded to numbers using the following mapping:

‘A’ -> 1.
‘B’ -> 2.
….
‘Z’ -> 26.

Given an encoded message containing digits, determine the total number of ways to decode it.

For example, Given encoded message “12”, it could be decoded as “AB” (1 2) or “L” (12).

The number of ways decoding “12” is 2.

解析：

依旧是fibonacci序列，只是每一个位置的dp值会随字符串s中对应的字符而变化，加上一些条件判断即可。

令dp[i]表示字符串的前i个字符的可能的编码方式的数量。那么递推公式如下：

如果前1个数字为0，那么当前数字只能从前2个数字跳转 dp[i] = dp[i-2]

如果前面数字为1或者前面数字为2且当前数字在1~6之间，说明都可以进行2种编码，而如果这种情况不存在，那么当前数字只能从前1个数字跳转 dp[i] = dp[i-1]。

此题可用滚动数组优化,3个变量（i，i-1，i-2），只需要2个临时量就可以保存

class Solution {
public:
int numDecodings(string s) {
    if (s.empty() || s[0] == '0') return 0;
    int prev = 0;    //dp[0]
    int cur = 1;    //dp[1]，可保证转码成功，所以为1
    for (size_t i = 1; i <= s.size(); ++i) {
        if (s[i-1] == '0') cur = 0;
    //dp[i] = dp[i-2]，所以先将dp[i-1]清0.
        if (i < 2 || !(s[i - 2] == '1' ||
                 (s[i - 2] == '2' && s[i - 1] <= '6')))
            prev = 0;
    //dp[i] = dp[i-1]，所以先将dp[i-2]清0.
        int tmp = cur;
        cur = prev + cur;//维护结果
        prev = tmp;//更新dp[i-2]
    }
    return cur;
}
};

			tmp1 = res;
			tmp2 = pre;
			res = res + prepre;
			pre = tmp1;
			prepre = tmp2;
		}
		return res;

	public static int getNum3(int n) {
		if (n < 1) {
			return 0;
		}
		if (n == 1 || n == 2) {
			return n;
		}
		int[][] base = { { 1, 1 }, { 1, 0 } };
		int[][] res = matrixPower(base, n - 2);
		return 2 * res[0][0] + res[1][0];
	}

	public static int[][] matrixPower(int[][] m, int p) {
		int[][] res = new int[m.length][m[0].length];
		for (int i = 0; i < res.length; i++) {
			res[i][i] = 1;
		}
		int[][] tmp = m;
		for (; p != 0; p >>= 1) {
			if ((p & 1) != 0) {
				res = muliMatrix(res, tmp);
			}
			tmp = muliMatrix(tmp, tmp);
		}
		return res;
	}

	public static int[][] muliMatrix(int[][] m1, int[][] m2) {
		int[][] res = new int[m1.length][m2[0].length];
		for (int i = 0; i < m1.length; i++) {
			for (int j = 0; j < m2[0].length; j++) {
				for (int k = 0; k < m2.length; k++) {
					res[i][j] += m1[i][k] * m2[k][j];
				}
			}
		}
		return res;
	}

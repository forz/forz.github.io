LeetCode 459. Repeated Substring Pattern#

给定一个非空字符串检查，如果它可以通过取其子字符串并将子字符串的多个副本附加在一起构造。 您可以假定给定字符串仅由小写英文字母组成，其长度不超过10000。

解析：设str长度为len，重复的子串长度为k，则如果真的由连续多个长度为k的子串重复构成str，那么在对str求next时，由于连续对称性，会从next[k+1]开始，1,2,3…地递增，直到next[len]=len-k，且(len-k)%k==0，表示有整数个k

class Solution {
public:
bool repeatedSubstringPattern(string str)
{
int len = str.length();
int next[len+1];
next[0] = -1;
int j = 0, k = -1;
while( j!=len )
//要一直求到next[len]而不是next[len-1]，是因为next[len-1]只是表示前len-1个字母的内部对称性，而没有考虑到最后一个字母即str[len-1]
{
    while( k!=-1 && str[j]!=str[k] )k = next[k];
        next[++j] = ++k;
}
return next[len]&&next[len]%(len-next[len])==0;

}};

LeetCode 28. Implement strStr()#

Implement strStr().

Returns the index of the first occurrence of needle in haystack, or -1 if needle is not part of haystack.

解析：

标准KMP算法，先对匹配串求next，然后和模板串匹配。求next的过程中可以进行一些优化。

class Solution {
public:
void getNext(int *next, string &needle) {
    int i = 0, j = -1;
    next[0] = -1;
    while (i < needle.length() - 1) {
        while (j != -1 && needle[i] != needle[j]) j = next[j];//next数组表示每个元素的失配位置
        ++i; ++j;
        //特殊情况，这里即为优化之处。考虑下AAAAB, 防止4个A形成0123在匹配时多次迭代。
        //当前是i发生失配，如果i和j相等，那么j怎么可能合适？所以直接找j的失配位即可
        if (needle[i] == needle[j]) next[i] = next[j]; 
        else next[i] = j;
    }
}
int strStr(string haystack, string needle) {
    if (haystack.empty()) return needle.empty() ? 0 : -1;
    if (needle.empty()) return 0;
    int next[needle.length() + 1];//长度要比字符串大1
    getNext(next, needle);//求得needle的next值
    int i = 0, j = 0;
    while (i != haystack.length()) {//kmp匹配（i代表模板串，j代表匹配串）
        while (j != -1 && haystack[i] != needle[j]) j = next[j];
        ++i; ++j;
        if (j == needle.length()) return i - j;//i是当前母串的位置，j现在是匹配串的长度
    }
    return -1;
}
};

class Solution {
public:
int strStr(string haystack, string needle) {
    int i, j;
    for (i = j = 0; i < haystack.size() && j < needle.size();) {
        if (haystack[i] == needle[j]) {
            ++i; ++j;
        } else {
            i -= j - 1; j = 0;
        }
    }
    return j != needle.size() ? -1 : i - j;
}
};

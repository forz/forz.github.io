LeetCode 341. Flatten Nested List Iterator#

Each element is either an integer, or a list – whose elements may also be integers or other lists.

Example 1:

Given the list [[1,1],2,[1,1]],

By calling next repeatedly until hasNext returns false, the order of elements returned by next should be: [1,1,2,1,1].

Example 2:

Given the list [1,[4,[6]]],

By calling next repeatedly until hasNext returns false, the order of elements returned by next should be: [1,4,6].

/**
 * // This is the interface that allows for creating nested lists.
 * // You should not implement it, or speculate about its implementation
 * class NestedInteger {
 *   public:
 *     // Return true if this NestedInteger holds a single integer, rather than a nested list.
 *     bool isInteger() const;
 *
 *     // Return the single integer that this NestedInteger holds, if it holds a single integer
 *     // The result is undefined if this NestedInteger holds a nested list
 *     int getInteger() const;
 *
 *     // Return the nested list that this NestedInteger holds, if it holds a nested list
 *     // The result is undefined if this NestedInteger holds a single integer
 *     const vector<NestedInteger> &getList() const;
 * };
 */ 

class NestedIterator {
public:
NestedIterator(vector<NestedInteger> &nestedList) {
    for (int i = nestedList.size() - 1; i >= 0; --i) 
        //构造的时候直接全部倒序压入栈
        s.push(nestedList[i]);
}
int next() {//取出
    NestedInteger t = s.top(); s.pop();
    return t.getInteger();
}
bool hasNext() {
    while (!s.empty()) {//遍历栈
        NestedInteger t = s.top(); //循环停止条件是栈为空，返回false，
        if (t.isInteger()) return true;//如果栈顶元素是整数，直接返回true，
        s.pop();//如果不是整数，那么移除栈顶元素，
        for (int i = t.getList().size() - 1; i >= 0; --i) {
            //遍历这个取出的list，还是从后往前压入栈
            s.push(t.getList()[i]);
        }
    }
    return false;
}
private:
    stack<NestedInteger> s;
};

/**
 * Your NestedIterator object will be instantiated and called as such:
 * NestedIterator i(nestedList);
 * while (i.hasNext()) cout << i.next();
 */

LeetCode 385. Mini Parser#

/**
 * // This is the interface that allows for creating nested lists.
 * // You should not implement it, or speculate about its implementation
 * class NestedInteger {
 *   public:
 *     // Constructor initializes an empty nested list.
 *     NestedInteger();
 *
 *     // Constructor initializes a single integer.
 *     NestedInteger(int value);
 *
 *     // Return true if this NestedInteger holds a single integer, rather than a nested list.
 *     bool isInteger() const;
 *
 *     // Return the single integer that this NestedInteger holds, if it holds a single integer
 *     // The result is undefined if this NestedInteger holds a nested list
 *     int getInteger() const;
 *
 *     // Set this NestedInteger to hold a single integer.
 *     void setInteger(int value);
 *
 *     // Set this NestedInteger to hold a nested list and adds a nested integer to it.
 *     void add(const NestedInteger &ni);
 *
 *     // Return the nested list that this NestedInteger holds, if it holds a nested list
 *     // The result is undefined if this NestedInteger holds a single integer
 *     const vector<NestedInteger> &getList() const;
 * };
 */
//嵌套链表可以看成由链表组成的树
class Solution {  
public:  
NestedInteger deserialize(string s) {  
    if(s[0] != '[') return NestedInteger(stoi(s));
    //如果第一个不是 [  那么不存在嵌套链表，而且只有一个整数
    stack<NestedInteger*> stk;  //栈中最好存储指针，能够降低空间复杂度。
//字符串中括号存在层层嵌套的情况，所以需要借助栈来记录遍历路径上的每层NestedInteger。
    NestedInteger* ret = NULL;  //起点
    int idx = 0;  
    for(int i=0; i<s.size(); ++i) {  
        if(s[i] == '[') {  
          //遇到'['时，表明即将进入新一层链表，此时构造新的NestedInteger并进栈；
            stk.push(new NestedInteger());  //放入新的迭代器来存放链表
            if(!ret) ret = stk.top(); //如果起点为空，直接从stk中读取该迭代器作为根迭代器
            idx = i + 1;  //记录下一个数字应出现的位置
        }  
        else if(s[i] == ']' || s[i] == ',') {  
            if(idx != i)  
//如果不相等，那么说明需要将字符串转化为数字(stoi)，然后存到栈顶迭代器中
                stk.top()->add(NestedInteger(stoi(s.substr(idx, i-idx))));  
            if(s[i] == ']') {              
            //遇到']'时，表明该层链表结束，此时出栈并将该层链表通过add函数加入上层链表中。
                NestedInteger* cur = stk.top();  
                stk.pop();  
                if(!stk.empty()) stk.top()->add(*cur);  
            }  
            idx = i + 1;  //记录数字的起始位置
        }  
    }  

    return *ret;  
}  
};

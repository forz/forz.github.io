LeetCode 74. Search a 2D Matrix#

Write an efficient algorithm that searches for a value in an m x n matrix. This matrix has the following properties:

Integers in each row are sorted from left to right. The first integer of each row is greater than the last integer of the previous row.

解析：

因为每行从小到大排序，而且每行第一列元素大于上一行最后一位元素，所以可以直接把这个矩阵看作一维有序数组。找到左右边界，然后直接二分即可。坐标除以行和取余行来得到相应坐标。

常规二分查找，while条件一定要包含相等的情况，否则会丢解。如果while结束还没有返回true，说明没有正确结果

class Solution {
public:
bool searchMatrix(vector<vector<int> > &matrix, int target) {
    int n = matrix.size();//行
    if(n==0) return false;
    int m = matrix[0].size();//列
    if(m==0) return false;
    int l = 0, r = m * n - 1;//左右边界
    while (l <= r){//二分查找
        int mid = l + (r - l) / 2;
        if(matrix[mid / m][mid % m] == target) return true;
        else if (matrix[mid / m][mid % m] < target)
            l = mid + 1;
        else 
            r = mid - 1;
    }
    return false;//二分查找没找到，返回false
}
};

LeetCode 240. Search a 2D Matrix II#

Write an efficient algorithm that searches for a value in an m x n matrix. This matrix has the following properties:

Integers in each row are sorted in ascending from left to right.

Integers in each column are sorted in ascending from top to bottom.

解析：从左到右有序，从上到下有序。每次都选取右上角数字，如果该值比目标大，那么可以排除最右端一列，如果该值比目标小，那么可以排除最顶端一行。根据判断结果维护新的右上角。不同的判断结果都可以消去一部分备选结果。

class Solution {
public:
bool searchMatrix(vector<vector<int>>& matrix, int target) {
int m = matrix.size();
if (m == 0) return false;//健壮性
int n = matrix[0].size();
int i = 0, j = n - 1;
while (i < m && j >= 0) {//保证不越界
    if (matrix[i][j] == target)//找到结果
        return true;
    else if (matrix[i][j] > target) {//排除当前列
        j--;
    } else //排除当前行
        i++;
}
return false;
}
};

LeetCode 378. Kth Smallest Element in a Sorted Matrix#

Given a n x n matrix where each of the rows and columns are sorted in ascending order, find the kth smallest element in the matrix.

Note that it is the kth smallest element in the sorted order, not the kth distinct element.

Example:

matrix = [
   [ 1,  5,  9],
   [10, 11, 13],
   [12, 13, 15]
],

k = 8,

return 13.

Note:

You may assume k is always valid, 1 ≤ k ≤ n2.

解析：由题可以判断左上角和右下角分别是最小值和最大值。可以用二分法，数的范围就是最小值到最大值之间。每找到一个mid，就遍历矩阵，查找比其小的元素个数。判断是否合格。

class Solution {//二分法
public:
int kthSmallest(vector<vector<int>>& matrix, int k) 
{
    int size = matrix.size(), l = matrix[0][0], r = matrix[size-1][size-1];
    while(l < r)
    {
        int smaller = 0, m = l+((r-l)>>1);//
        for(int i = 0; i < size; ++i) //扫描每一行，查找比小于等于M的元素个数
            smaller += upper_bound(matrix[i].begin(), matrix[i].end(), m)-matrix[i].begin();
        smaller<k? l=m+1 : r=m;//如果比k小，说明m不够大
    }
    return r;
}
};

class Solution {
public:
struct compare//构建最小优先队列
{
bool operator()(const pair<int,pair<int, int> >& a, const pair<int,pair<int, int> >& b)
{
    return a.first>b.first;
}
};
int kthSmallest(vector<vector<int>>& arr, int k) {
    int n=arr.size(),m=arr[0].size();
    priority_queue< pair<int,pair<int, int> >, vector<pair<int, pair<int, int> > >, compare > p;
    for(int i=0;i<n;i++)
    p.push(make_pair(arr[i][0],make_pair(i,0)));
    int x=k -1,ans;
    while(x--)
    {
        int ans=p.top().first;
        int i=p.top().second.first;
        int j=p.top().second.second;
        p.pop();
        if(j!=m-1)
        p.push(make_pair(arr[i][j+1],make_pair(i,j+1)));
    }
    return p.top().first;
}
};

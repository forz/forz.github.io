LeetCode 36. Valid Sudoku#

The Sudoku board could be partially filled, where empty cells are filled with the character ‘.’.

解析：检查九宫格是否有效，只需要检查每行每列和每个子九宫格是否存在重复数字即可。创建三个map来进行哈希映射。为了简化代码，可以将3种检查放在一个循环中。

class Solution {
public:
bool isValidSudoku(vector<vector<char> > &board) {//数独：9行9列9宫
    for(int i = 0; i < 9; i ++)//枚举第一维（分别表示行号，列号，九宫格号）
    {
        unordered_map<char, bool> m1;   //check i_th row
        unordered_map<char, bool> m2;   //check i_th column
        unordered_map<char, bool> m3;   //check i_th subgrid
        for(int j = 0; j < 9; j ++)//枚举第二维（分别表示列号，行号，九宫格中的数字号(共9个数字)）
        {
            if(board[i][j] != '.')//检查每行
            {
                if(m1[board[i][j]] == true)
                    return false;
                m1[board[i][j]] = true;
            }
            if(board[j][i] != '.')//检查每列（i和j互换位置，就可以和检查行同步进行）
            {
                if(m2[board[j][i]] == true)
                    return false;
                m2[board[j][i]] = true;
            }
            //对于每三个九宫格行号增3；对于单个九宫格，每三个格点行号增1。
            //因此第i个九宫格的第j个格点的行号可表示为i/3*3+j/3

            //对于下个九宫格列号增3，循环周期为3；对于单个九宫格，每个格点行号增1，周期也为3。
            //周期的数学表示就是取模运算mod。因此第i个九宫格的第j个格点的列号可表示为i%3*3+j%3
            if(board[i/3*3+j/3][i%3*3+j%3] != '.')
            {
                if(m3[board[i/3*3+j/3][i%3*3+j%3]] == true)
                    return false;
                m3[board[i/3*3+j/3][i%3*3+j%3]] = true;
            }
        }
    }
    return true;
}
};

LeetCode 37. Sudoku Solver#

Write a program to solve a Sudoku puzzle by filling the empty cells.

Empty cells are indicated by the character ‘.’.

You may assume that there will be only one unique solution.

解析：填写九宫格。每个空格可以填9种数字，回溯法进行暴力搜索。每次填写前都要判断当前数字是否有效（上题），然后进行dfs。

class Solution {
public:
bool solveSudoku(vector<vector<char> > &board) {
    for (int i = 0; i < 9; ++i)
        for (int j = 0; j < 9; ++j) {
            if (board[i][j] == '.') {
                for (int k = 0; k < 9; ++k) {
                    board[i][j] = '1' + k;
                    if (isValid(board, i, j) && solveSudoku(board))
                        return true;
                    board[i][j] = '.';//回溯
                }
                return false;
            }
        }
    return true;
}
private:
// 检查 (x, y) 是否合法
bool isValid(const vector<vector<char> > &board, int x, int y) {
    int i, j;
    for (i = 0; i < 9; i++) // 检查 y 列
        if (i != x && board[i][y] == board[x][y])
            return false;
    for (j = 0; j < 9; j++) // 检查 x 行
        if (j != y && board[x][j] == board[x][y])
            return false;
    //检查(x,y)所在的子九宫格的9个数字
    for (i = 3 * (x / 3); i < 3 * (x / 3 + 1); i++)//横坐标范围
        for (j = 3 * (y / 3); j < 3 * (y / 3 + 1); j++)//纵坐标范围
            if ((i != x || j != y) && board[i][j] == board[x][y])
                return false;
    return true;
}
};

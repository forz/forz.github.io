LeetCode 88. Merge Sorted Array#

Given two sorted integer arrays nums1 and nums2, merge nums2 into nums1 as one sorted array.

Note:

You may assume that nums1 has enough space (size that is greater or equal to m + n) to hold additional elements from nums2. The number of elements initialized in nums1 and nums2 are m and n respectively.

解析：

合并两个数组（或字符串时），如果从前往后复制每个数字，需要重复移动数字多次，那么我们可以考虑从后往前复制，这样就能减少移动的次数，从而提高效率。

class Solution {
public:
void merge(vector<int>& nums1, int m, vector<int>& nums2, int n) {
    int i = m - 1, j = n - 1, tar = m + n - 1;//将nums2放到nums1中
    while (j >= 0) {//直到把nums2完全插入
        nums1[tar--] = (i >= 0 && nums1[i] > nums2[j] )? nums1[i--] : nums2[j--];
    }
}
};

LeetCode 21. Merge Two Sorted Lists#

Merge two sorted linked lists and return it as a new list. The new list should be made by splicing together the nodes of the first two lists.

解析：链表中的merge不能从后往前进行，所以只能按照常规来处理，采用递归的方法。

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode *mergeTwoLists(ListNode *l1, ListNode *l2) 
    {
        if (l1 == NULL) return l2;
        if (l2 == NULL) return l1;
        ListNode *ret = NULL;
        if (l1->val < l2->val)
        {
            ret = l1;
            ret->next = mergeTwoLists(l1->next, l2);
        }
        else
        {
            ret = l2;
            ret->next = mergeTwoLists(l1, l2->next);
        }
        return ret;
    }
};

class Solution {
public:
ListNode *mergeTwoLists(ListNode *l1, ListNode *l2) {
    ListNode dummy(INT_MIN);//创建伪头节点
    ListNode *tail = &dummy;
    while (l1 && l2) {
        if (l1->val < l2->val) {
            tail->next = l1;
            l1 = l1->next;
        } else {
            tail->next = l2;
            l2 = l2->next;
        }
        tail = tail->next;
    }
    tail->next = l1 ? l1 : l2;//将剩下的部分直接连接到tail上
    return dummy->next;//返回真正的头节点
}
};

LeetCode 23. Merge k Sorted Lists#

Merge k sorted linked lists and return it as one sorted list. Analyze and describe its complexity.

解析：多个有序列表同时合并，利用堆排序优化，迅速找到最小值。

/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
struct cmp {
    bool operator() (ListNode *a,ListNode *b) {//构建最小队列
        return a->val > b->val;
    }
};
ListNode *mergeKLists(vector<ListNode *> &lists) {
    priority_queue<ListNode*,vector<ListNode*>,cmp> queue;//堆排序优化
    //因为要自定义比较仿函数，所以要填写正确的容器类型
    for(int i=0;i<lists.size();i++) {
        if(lists[i]!=NULL) {
            queue.push(lists[i]);
        }
    }
    ListNode *head=NULL,*prev=NULL,*temp;
    while(!queue.empty()) {
        temp=queue.top();//temp为当前最小节点
        queue.pop();
        if(prev==NULL) {//第一次放入节点
            head=temp;
        }else {
            prev->next=temp;
        }
        prev=temp;//保存前驱节点
        if(temp->next!=NULL) {//将最小节点的后继放入优先队列
            queue.push(temp->next);
        }
    }
    return  head;
}  
};

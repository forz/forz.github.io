LeetCode102. Binary Tree Level Order Traversal

Given a binary tree, return the level order traversal of its nodes’ values. (ie, from left to right, level by level).

For example: Given binary tree [3,9,20,null,null,15,7],

  3
 / \
9  20
  /  \
 15   7

return its level order traversal as:

[
  [3],
  [9,20],
  [15,7]
]

解析：此题要求将二叉树层序遍历，使用队列。要求分层输出，用2个队列换着保存。

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
vector<vector<int>> levelOrder(TreeNode* root) {
    vector<vector<int>>result; //存储结果 
    if (!root)  //空树
        return result;  
    queue<TreeNode*> vec;  //第一个队列，保存每层节点
    vec.push(root);  //放入根节点
    while (!vec.empty()) {  
        queue<TreeNode*> tmp_vec;  //第二个队列，保存下一层节点
        vector<int> tmp_result;  //保存当前层的遍历结果
        while (!vec.empty()) { //BFS
            TreeNode* node = vec.front(); //读取根节点 
            vec.pop();  
            if (node->left)  //把下一层节点放入队列2
                tmp_vec.push(node->left);  
            if (node->right)  
                tmp_vec.push(node->right);  
            tmp_result.push_back(node->val); //保存当前层结果
        }  
        vec = tmp_vec;  //维护第一个队列
        result.push_back(tmp_result);  //维护结果
    }  
    return result;  
}
};

#107. Binary Tree Level Order Traversal II Given a binary tree, return the bottom-up level order traversal of its nodes’ values. (ie, from left to right, level by level from leaf to root).

For example:
Given binary tree [3,9,20,null,null,15,7],

  3
 / \
9  20
  /  \
 15   7

return its bottom-up level order traversal as:

[
  [15,7],
  [9,20],
  [3]
]

解析：同上题，最后结果进行翻转即可。

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution  
{  
public:  
vector<vector<int>> levelOrderBottom(TreeNode* root)  //BFS，保存每层的遍历结果
{  
    vector<vector<int>>result; //存储结果 
    if (!root)  //空树
        return result;  
    queue<TreeNode*> vec;  //第一个队列，保存每层节点
    vec.push(root);  //放入根节点
    while (!vec.empty()) {  
        queue<TreeNode*> tmp_vec;  //第二个队列，保存下一层节点
        vector<int> tmp_result;  //保存当前层的遍历结果
        while (!vec.empty()) { //BFS
            TreeNode* node = vec.front(); //读取根节点 
            vec.pop();  
            if (node->left)  //把下一层节点放入队列2
                tmp_vec.push(node->left);  
            if (node->right)  
                tmp_vec.push(node->right);  
            tmp_result.push_back(node->val); //保存当前层结果
        }  
        vec = tmp_vec;  //维护第一个队列
        result.push_back(tmp_result);  //维护结果
    }  
    //对层次遍历结果进行翻转  
    reverse(result.begin(), result.end());  
    return result;  
}  
};

103. Binary Tree Zigzag Level Order Traversal

Given a binary tree, return the zigzag level order traversal of its nodes’ values. (ie, from left to right, then right to left for the next level and alternate between).

For example:

Given binary tree [3,9,20,null,null,15,7],

  3
 / \
9  20
  /  \
 15   7

return its zigzag level order traversal as:

[
  [3],
  [20,9],
  [15,7]
]

解析：和前两题相同，只是每层扫描的时候需要先判断扫描方向。可以把之前的两个队列简化为一个队列，条件是先求出上一层的节点个数。然后只从队列取出固定个数个节点。

/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode(int x) : val(x), left(NULL), right(NULL) {}
 * };
 */
class Solution {
public:
vector<vector<int> > zigzagLevelOrder(TreeNode* root) {
if (root == NULL) //empty tree 
    return vector<vector<int> > ();
vector<vector<int> > result;
queue<TreeNode*> nodesQueue;
nodesQueue.push(root);
bool leftToRight = true;
while ( !nodesQueue.empty()) {
    int size = nodesQueue.size();//当前层的节点总数
    vector<int> row(size);//保存每层结果
    for (int i = 0; i < size; i++) {//遍历当前层所有节点
        TreeNode* node = nodesQueue.front();
        nodesQueue.pop();
        int index = (leftToRight) ? i : (size - 1 - i);//根据当前方向判断该节点在结果数组中的位置。（因为数组长度size已知）
        row[index] = node->val;//放入数组
        if (node->left)    nodesQueue.push(node->left);
        if (node->right)   nodesQueue.push(node->right);
    }
    leftToRight = !leftToRight;//控制下一层方向
    result.push_back(row);
}
return result;
}
};

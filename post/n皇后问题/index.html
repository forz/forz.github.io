LeetCode 51. N-Queens

The n-queens puzzle is the problem of placing n queens on an n×n chessboard such that no two queens attack each other.

Given an integer n, return all distinct solutions to the n-queens puzzle.

Each solution contains a distinct board configuration of the n-queens’ placement, where ‘Q’ and ‘.’ both indicate a queen and an empty space respectively.

For example, There exist two distinct solutions to the 4-queens puzzle:

[
 [".Q..",  // Solution 1
  "...Q",
  "Q...",
  "..Q."],

 ["..Q.",  // Solution 2
  "Q...",
  "...Q",
  ".Q.."]
]

解析：

标准回溯法，因为按行枚举，所以判断位置是否合法需要看当前列，副对角线，主对角线是否有其他皇后，这要多加一次扫描。但可以用vis数组结合数学方法消除这次扫描。

vis数组有三维，分别表示列，副对角线，主对角线,可以不必循环求是否安全，直接看数组中相应的元素

i表示列 c+i副对角线 c-i+n主对角线

class Solution
{
vector<vector<string> > vvs;
public:
vector<vector<string> > solveNQueens(int n)
{
    vector<int> vi(n);//保存每行皇后所在的列（因为按行放置，所以肯定不会横向攻击）
    vector<vector<int> > vis(3,vector<int>(n+n));
    solveNQueens(vi, n, 0,vis);
    return vvs;
}
private:
void solveNQueens(vector<int> &vi, int n, int c, vector<vector<int> >&vis)
{//vi表示放置皇后的列数组    n表示N*N的棋盘     c表示当前放置到c行
    if(c == n)//到最后一行，递归终止
    {
        vector<string> vs(n, string(n, '.'));//构建图
        for(int i = 0; i < n; ++ i)//按行放置每个皇后
            vs[i][vi[i]] = 'Q';
        vvs.push_back(vs);//放入结果
        return;
    }
    for(int i = 0; i < n; ++i)//枚举每一列
        if(!vis[0][i]&&!vis[1][c+i]&&!vis[2][c-i+n])//判断是否合适
        {
            vi[c]=i;
            vis[0][i]=vis[1][c+i]=vis[2][c-i+n]=1;
            solveNQueens(vi, n, c + 1,vis);//进行dfs
            vis[0][i]=vis[1][c+i]=vis[2][c-i+n]=0;//回溯
        }

}
};

Now, instead outputting board configurations, return the total number of distinct solutions.

class Solution
{
public:
int totalNQueens(int n) 
{
    vector<vector<int> > vis(3,vector<int>(n+n));
    int cnt=0;
    solveNQueens(cnt, n, 0,vis);
    return cnt;
}
private:
void solveNQueens(int &cnt, int n, int c, vector<vector<int> >&vis)
{//vi表示放置皇后的列数组    n表示N*N的棋盘     c表示当前放置到c行
    if(c == n)//到最后一行
        cnt++;
    for(int i = 0; i < n; ++i)//枚举每一列
        if(!vis[0][i]&&!vis[1][c+i]&&!vis[2][c-i+n])//判断是否合适
        {//vis数组有三维，分别表示列，副对角线，主对角线,可以不必循环求是否安全，直接看数组
        //i表示列       c+i副对角线       c-i+n主对角线
            vis[0][i]=vis[1][c+i]=vis[2][c-i+n]=1;
            solveNQueens(cnt, n, c + 1,vis);//进行dfs
            vis[0][i]=vis[1][c+i]=vis[2][c-i+n]=0;//回溯
        }

}
};
